#!/bin/bash

# Download Japanese fonts
echo "Downloading NotoSans Japanese fonts..."
wget -q "https://noto-website-2.storage.googleapis.com/pkgs/NotoSansCJKjp-hinted.zip" -O NotoSansCJKjp.zip

# Unzip and install
echo "Installing fonts..."
unzip -q NotoSansCJKjp.zip -d fonts_temp
mkdir -p ~/.local/share/fonts/  # Modern font directory
cp fonts_temp/*.otf ~/.local/share/fonts/

# Update font cache
echo "Updating font cache..."
fc-cache -f ~/.local/share/fonts/ > /dev/null

# Cleanup
rm -rf NotoSansCJKjp.zip fonts_temp
echo "âœ… Japanese fonts installed successfully!"
